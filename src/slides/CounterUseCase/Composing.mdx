### CounterDisplay code

```jsx filename="CounterDisplay.jsx" highlight="[2]"
const CounterDisplay = () => {
  const { value } = useCounter();

  return <BaseCounterDisplay value={value} />;
};
```

### CounterButton code

```jsx filename="CounterButton.jsx" highlight="[2]"
const CounterButton = ({ children, mode, step = 1 }) => {
  const { value, onChange, min, max } = useCounter();
  const props = { mode, step, value, onChange, min, max };

  return <BaseCounterButton {...props}>{children}</BaseCounterButton>;
};
```

# Composing
